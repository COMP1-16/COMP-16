int fat(int n) {
    if (n <= 1) {
        return 1;
    }
    // ERRO: Erro de digitação na chamada recursiva. Chamou 'fato' ao invés de 'fat'.
    return n * fato(n - 1);
}

    printf("Tentando rodar com nome errado na chamada recursiva...\n");
    fat(5);
    return 0;
}
