int main() {
    int i = 0;
    int soma = 0;
    while (i < 4) {
        {
            if (i % 2 == 0) {
                i = i + 1;
                continue;
            } else {
                soma = soma + i;
            }
        }
        i = i + 1;
    }
    return soma;  // ESPERADO: 1+3 = 4
}
