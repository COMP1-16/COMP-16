// IMPORTANTE: no while, nao ha incremento automatico como no for.
// O incremento precisa vir ANTES do continue, senao gera loop infinito.
int main() {
    int i = 0;
    int contador = 0;
    while (i < 4) {
        i = i + 1;  // incremento antes do continue
        if (i == 2) {
            continue;
        }
        contador = contador + 1;
    }
    return contador;  // ESPERADO: 3 (conta i=1,3,4; pula i=2)
}
