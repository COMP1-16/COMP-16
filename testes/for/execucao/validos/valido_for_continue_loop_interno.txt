int func() {
    int i, j;
    int total = 0;
    for (i = 0; i < 2; i = i + 1) {
        for (j = 0; j < 3; j = j + 1) {
            if (j == 1) {
                continue;  // afeta so o for interno
            }
            total = total + 1;
        }
    }

    return total;  // ESPERADO: 2 * 2 = 4
}

int result = func();

printf("%d\n", result);
